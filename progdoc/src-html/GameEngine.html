183    private void eat(final Command pCommand) {
184
185        if (!pCommand.hasSecondWord()) {
186            this.aGUI.println("Eat what?");
187            return;
188        }
189
190        String vItemName = pCommand.getSecondWord();
191        if (this.aPlayer.getInventory().hasItem(vItemName) == false) {
192            this.aGUI.println("You don't have that item.");
193            return;
194        }
195
196        Item vItem = this.aPlayer.getCurrentRoom().getItem(vItemName);
197        if (this.aPlayer.getCurrentRoom().getItem("couscous") == null
198                && this.aPlayer.getCurrentRoom().getItem("magic-msemmen") == null) {
199            this.aGUI.println("Are you crazy? You can't eat that!");
200            return;
201        } else if (this.aPlayer.getCurrentRoom().getItem("couscous") != null) {
202            this.aPlayer.eat(vItemName);
203            this.aGUI.println("Yummy, nothing better than some good food");
204            return;
205        } else if (this.aPlayer.getCurrentRoom().getItem("magic-msemmen") != null) {
206            this.aPlayer.eat(vItemName);
207            this.aGUI.println("Something mystical is happening to your body");
208            this.aPlayer.setMaxWeight(this.aPlayer.getMaxWeight() * 2);
209            return;
210        }
211    }

213    /**
214     * Change la room oÃ¹ se trouve le joueur
215     * 
216     * @param pCommand La commande
217     */
218    public void goRoom(final Command pCommand) {
219        if (!CommandWords.isDirection(pCommand.getSecondWord())) {
220            this.aGUI.println("Where even is that?");
221            return;
222        }
223
224        if (pCommand.hasSecondWord() == false) {
225            this.aGUI.println("Go Where ?");
226            return;
227        }
228
229
230        if (this.aPlayer.getCurrentRoom() instanceof TransporterRoom) {
231            TransporterRoom vTransporter = (TransporterRoom) this.aPlayer.getCurrentRoom();
232            if (this.aIsInTestMode && !this.aFakeRandomRoom.isEmpty()) {
233                ArrayList<Room> vRooms = new ArrayList<Room>(vTransporter.getExits());
234                for (Room pRoom : vRooms) {
235                    String vRoomName = pRoom.getDescription().toLowerCase();
236                    if (this.aFakeRandomRoom.contains(vRoomName.toLowerCase())) {
237                        this.aPlayer.setCurrentRoom(pRoom);
238                        this.printLocationInfo();
239                        return;
240                    }
241                }
242                this.printLocationInfo();
243                return;
244            }
245            this.aPlayer.setCurrentRoom(vTransporter.getRandomExit());
246            this.aPlayer.getPreviousRooms().clear();
247            return;
248        }
249
250        String vDirection = pCommand.getSecondWord();
251        Room vNextRoom = this.aPlayer.getCurrentRoom().getExit(vDirection);
252
253        if (vNextRoom == null) {
254            this.aGUI.println("There is no door !");
255            return;
256        }
257
258        this.aPlayer.goRoom(vDirection);
259        this.printLocationInfo();
260    }

304        } else {

357    /**
358     * Charges the Beamer
359     * 
360     * @param pCommand The command to execute
361     */
362    public void charge(final Command pCommand) {
363        if (!pCommand.hasSecondWord()) {
364            this.aGUI.println("Charge what?");
365            return;
366        }
367
368        String vItemName = pCommand.getSecondWord();
369
370        if (!this.aPlayer.getInventory().hasItem(vItemName)) {
371            this.aGUI.println("You don't have " + vItemName + ".");
372            return;
373        }
374
375        if (!(this.aPlayer.getInventory().getItem(vItemName) instanceof Beamer)) {
376            this.aGUI.println("You can't charge " + vItemName + " !");
377            return;
378        }
379
380        Item vItem = this.aPlayer.getInventory().getItem(vItemName);
381        Beamer vBeamer = (Beamer) vItem;
382
383        if (vBeamer.isCharged()) {
384            this.aGUI.println("You can't charge " + vItemName + " ! \n It is already charged.");
385            return;
386        }
387
388        this.aPlayer.charge(this.aPlayer.getCurrentRoom(), vBeamer);
389        this.aGUI.println("You charged " + vItemName + ".");
390
391    }

429    /**
430     * Transform the input into a readable command and interprets it.
431     * 
432     * @param pCommandLine
433     */
434    public void interpretCommand(final String pCommandLine) {
435        this.aGUI.println("> " + pCommandLine);
436        Command vCommand = this.aParser.getCommand(pCommandLine);
437
438        if (vCommand.isUnknown()) {
439            this.aGUI.println("I don't know what you mean...");
440            return;
441        }
442
443        String vCommandWord = vCommand.getCommandWord();
444        vCommandWord = vCommandWord.toLowerCase();
445        switch (vCommandWord) {
446            case "quit":
447                this.endGame();
448                break;
449            case "help":
450                this.printHelp();
451                break;
452            case "look":
453                this.look();
454                break;
455            case "eat":
456                this.eat(vCommand);
457                break;
458            case "test":
459                this.test(vCommand);
460                break;
461            case "back":
462                this.back(vCommand);
463                break;
464            case "take":
465                this.take(vCommand);
466                break;
467            case "drop":
468                this.drop(vCommand);
469                break;
470            case "go":
471                this.goRoom(vCommand);
472                break;
473            case "items":
474                this.printInventory(vCommand);
475                break;
476            case "charge":
477                this.charge(vCommand);
478                break;
479            case "fire":
480                this.fire(vCommand);
481                break;
482            case "alea":
483                this.alea(vCommand);
484                break;
485            default:
486                this.aGUI.println("I don't know what you mean...");
487                break;
488        }
489    }

517    /**
518     * Test method
519     */
520    private void test(final Command pCommand) {
521        this.aIsInTestMode = true;
522        if (pCommand.hasSecondWord() == false) { // We want to know if there is a second word
523            this.aGUI.println("Test what?");
524            return;
525        }
526        // We use the second word
527        String vTest = pCommand.getSecondWord();
528        try {
529            Scanner vScanner = new Scanner(new File("" + vTest + ".txt"));
530            this.aGUI.println("" + vTest + "is being tested...");
531            while (vScanner.hasNextLine()) {
532                interpretCommand(vScanner.nextLine());
533            }
534        } catch (FileNotFoundException pE) {// If the file is not existing
535            this.aGUI.println("File not found.");
536        }
537        this.aIsInTestMode = false;
538
539    }
